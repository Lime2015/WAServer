
drop database wadb;

create database wadb;

use wadb;

drop table man;

create table man (	
  assemblyman_id INT(10) NOT NULL primary key,
  assemblyman_name VARCHAR(30) NOT NULL,
  image_url VARCHAR(60),
  org_image_url VARCHAR(60),
  mod_dttm VARCHAR(60),
  party_id INT(10),
  party_name VARCHAR(60),
  local_constituency VARCHAR(60),
  update_tag INT(10)
);
